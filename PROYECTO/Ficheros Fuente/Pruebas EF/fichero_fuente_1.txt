let cadena string;

let x int;
print "Introduce un numero del 1 al 5 :";
input  x;

switch (x) {

    case 1:
        print "Tienes un 1!!";
        break;
    case 2:
    case 3:
    case 4:
        print "Tienes un 2, un 3 o un 4!!";
        break;
    default:
        print "Tienes un 5!!";
}

print "Escribe tu nombre: ";
input cadena;
print "Hola: ";
print cadena;

print "Escribe tu edad: ";
input edad;
